10. Calender Printer




import calendar 
print("\nCalender Programme\n  ")
print("Do you want to Print Calender for Just 1 mounth(1) or Whole Year(2) ?? :")
d = int(input())
if d == 1:

    y = int(input("Enter the Year :\n "))
    m = int(input("Enter the month :\n "))

    c = calendar.month(y , m)
    print("\n"+c+"\n")

elif d == 2:
    y = int(input("Enter the Year : "))
    i = 0
    while (i < 12):
        i = i+1
        c = calendar.month(y ,i )
        print("\n"+c+"\n")
print("Thank You For Using")
